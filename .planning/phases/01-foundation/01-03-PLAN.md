   ```python
   from aiogram.types import Message
   from typing import Optional

   def validate_message(message: Message) -> tuple[bool, Optional[str]]:
       """Returns (is_valid, rejection_reason)."""
       if message.forward_from or message.forward_from_chat:
           return False, "forwarded_message"
       if not message.photo:
           if message.document:
               return False, "document_not_photo"
           return False, "no_photo"
       return True, None
   ```

5. Create `shifttracker/bot/router.py`:
   ```python
   from aiogram import Router, F
   from aiogram.types import Message
   from shifttracker.pipeline.stages.validate import validate_message
   from shifttracker.pipeline.queue import enqueue_message
   from shifttracker.pipeline.models import ProcessingContext
   from shifttracker.db.engine import async_session_factory
   from shifttracker.db.models import TelegramGroup, ProcessingLog
   from sqlalchemy import select, update

   router = Router(name="shift_photo_router")

   def build_source_link(chat_id: int, message_id: int) -> str:
       chat_id_abs = str(chat_id).lstrip("-").lstrip("100")
       return f"https://t.me/c/{chat_id_abs}/{message_id}"

   @router.message(F.photo)
   async def handle_photo(message: Message) -> None:
       """Silent handler — never sends responses to the group chat."""
       is_valid, reason = validate_message(message)
       if not is_valid:
           # Log skip and return silently
           async with async_session_factory() as session:
               log_entry = ProcessingLog(
                   update_id=message.model.update_id if hasattr(message, 'model') else 0,
                   message_id=message.message_id,
                   chat_id=message.chat.id,
                   status="SKIPPED",
                   reason=reason,
               )
               session.add(log_entry)
               await session.commit()
           return

       source_link = build_source_link(message.chat.id, message.message_id)

       # Look up group
       async with async_session_factory() as session:
           result = await session.execute(
               select(TelegramGroup).where(TelegramGroup.chat_id == message.chat.id)
           )
           group = result.scalar_one_or_none()

       ctx = ProcessingContext(
           update_id=0,  # Will be set from Update object in middleware
           message_id=message.message_id,
           chat_id=message.chat.id,
           sender_user_id=message.from_user.id if message.from_user else None,
           caption=message.caption,
           message_datetime=message.date,
           group_id=group.id if group else None,
           group_timezone=group.timezone if group else "Europe/Moscow",
           shift_start_hour=group.shift_start_hour if group else 6,
           shift_end_hour=group.shift_end_hour if group else 22,
           source_link=source_link,
       )
       await enqueue_message(ctx)

   @router.message(F.migrate_to_chat_id)
   async def handle_migration(message: Message) -> None:
       """Handle group -> supergroup migration by updating chat_id."""
       old_chat_id = message.chat.id
       new_chat_id = message.migrate_to_chat_id
       async with async_session_factory() as session:
           await session.execute(
               update(TelegramGroup)
               .where(TelegramGroup.chat_id == old_chat_id)
               .values(chat_id=new_chat_id)
           )
           await session.commit()
   ```

   IMPORTANT: The bot MUST NOT send any reply messages to the group. All handlers return None (silent processing per locked decision).

1. Create `shifttracker/pipeline/stages/deduplicate.py`:
   ```python
   from sqlalchemy.ext.asyncio import AsyncSession
   from sqlalchemy.dialects.postgresql import insert as pg_insert
   from sqlalchemy import insert, select
   from shifttracker.db.models import ProcessedUpdate, ShiftRecord
   from datetime import date
   from uuid import UUID

   async def check_duplicate(update_id: int, session: AsyncSession) -> bool:
       """Returns True if update_id was already processed (duplicate).
       Uses INSERT ... ON CONFLICT DO NOTHING pattern.
       For SQLite (tests): uses try/except IntegrityError fallback.
       For PostgreSQL (prod): uses ON CONFLICT DO NOTHING and checks rowcount.
       """
       try:
           # Try insert — if it succeeds, this is a new update
           stmt = insert(ProcessedUpdate).values(update_id=update_id)
           # For SQLite compatibility in tests
           result = await session.execute(stmt)
           await session.flush()
           return False  # Not a duplicate — insert succeeded
       except Exception:
           await session.rollback()
           return True  # Duplicate — constraint violation

   async def check_business_duplicate(
       employee_id: UUID,
       shift_date: date,
       session: AsyncSession,
   ) -> bool:
       """Returns True if a shift_record already exists for this employee+date."""
       result = await session.execute(
           select(ShiftRecord.id).where(
               ShiftRecord.employee_id == employee_id,
               ShiftRecord.shift_date == shift_date,
           )
       )
       return result.scalar_one_or_none() is not None
   ```
